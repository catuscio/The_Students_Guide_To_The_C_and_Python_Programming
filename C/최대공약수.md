//최대공약수 구하기

#include <stdio.h>

int gcd_by_for(int n, int m) {
	int gcd = 0;

	for (int i = 0; i <= n && i <= m; i++)
		if (n % i == 0 && m % i == 0)
			gcd = i;

	return gcd;
}

int gcd_by_while(int n, int m) {
	int gcd = 0;

	while (n != m) {
		if (n >= m)
			n -= m;
		else
			m -= n;
	}

	return gcd;
}

int gcd_by_euclidean(int n, int m) {
	int tmp = n % m;
	if (tmp == 0)
		return m;

	return gcd_by_euclidean(m, n);
}

int main() {
	int n, m;
	int gcd;

	scanf_s("%d %d", &n, &m);

	printf("%d %d", n, m);

	printf("for문을 통해 구한 %d와 %d의 최대공약수는 %d\n", n, m, gcd_by_for(n, m));
	printf("while문을 통해 구한 %d와 %d의 최대공약수는 %d\n", n, m, gcd_by_while(n, m));
	printf("유클리드 호제법을 통해 구한 %d와 %d의 최대공약수는 %d\n", n, m, gcd_by_euclidean(n, m));

	return 0;
}
